  13. Find the output of the following program

    int m = 100, n= 200;
    int fun1(int count);
    int fun2(int p);

    void main( )
    {
    int count;
    for(count = 1; count <= 5; ++count)
    printf(“%d”, fun1(count));
    }
    int fun1(int a)
    {
    int r, s;
    r = fun2(a);
    s = (r<100)? (m + r): n;
    return s;
    }
    int fun2(int a)
    {
    static int val = 1;
    val * = a;
    return(val);
    }( Marks: -0.66 )
    1.   101 102 104 106 124
    2.   101 102 106 108 110
    3.   101 102 106 124 200
      Explanation:
      Given m=100, n=200,

      for count = 1,
      fun1(1) => r=fun2(1) , int val is static, val = val * 1 => val =1; so r=1
      now s=(r<100)?(m+r):n; it will be always true and m+r will be executed. 100+1 = 101 will be printed

      for count = 2,
      fun1(2) => r=fun2(2) , int val is static, val = val * 2 => val =2; so r=2
      now s=(r<100)?(m+r):n; it will be always true and m+r will be executed. 100+2 = 102 will be printed

      for count = 3,
      fun1(3) => r=fun2(3) , int val is static, val = val * 3 => val =6; so r=6
      now s=(r<100)?(m+r):n; it will be always true and m+r will be executed. 100+6 = 106 will be printed

      for count = 4,
      fun1(4) => r=fun2(4) , int val is static, val = val * 4 => val =24; so r=24
      now s=(r<100)?(m+r):n; it will be always true and m+r will be executed. 100+24 = 124 will be printed

      for count = 5,
      fun1(5) => r=fun2(5) , int val is static, val = val * 5 => val =120; so r=120
      now s=(r<100)?(m+r):n; it will be false and n will be executed. n = 200 will be printed
      like wise it will execute for count =5 also and terminates after as condition becomes false. So output will be 101 102 106 124 200

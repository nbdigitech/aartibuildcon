  12. Consider the relational schemas:
    StudentInfo(StudentId, StudentName, Age, Sex)
    CourseInfo(CourseId, semester, Year)
    Enroll(StudentId, Course_Id, Section_Id)
    Which one of the following query finds the section(s) and their corresponding courses that has maximum enrollment among all sections in all courses in 4th semester 2016.( Marks: 0.00 )
    1.   with sec_enroll as(
      select E.Course_id, E.Section_id, Count(E.StudentId) as TotStudents
      from Enroll E, CourseInfo C
      where E.Course_Id = C.CourseId
      AND C.Semester = 4 and C.year = 2016
      Group by E.Course_id, E.Section_id)
      Select Course_id, Section_id
      From Sec_enroll
      Where TotStudents = (select max(totstudents) from sec_enroll);
    2.   with sec_enroll as(
      select E.Course_id, E.Section_id, Count(E.StudentId) as TotStudents
      from Enroll E, CourseInfo C
      where E.Course_Id = C.CourseId
      AND C.Semester = 4 and C.year = 2016
      Group by E.Course_id, E.Section_id)
      select Course_id, Section_id
      from Sec_enroll;
    3.   with sec_enroll as(
      select E.Course_id, E.Section_id, Count(E.StudentId) as TotStudents
      from Enroll E, CourseInfo C
      where E.Course_Id = C.CourseId
      AND C.Semester = 4 and C.year = 2016
      Group by E.Course_id, E.Section_id)
      Select Course_id, Section_id
      from Enroll
      where
      Count(StudentId) = (select max(totstudents) from Sec_enroll);
    4.   None of these
